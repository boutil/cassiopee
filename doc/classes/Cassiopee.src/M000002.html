    def computeHamming(pattern,hamming)
        pattern = pattern.downcase
        nberr = 0
        (0..(self.length-1)).each do |c|
                if(pattern[c] != self[c])
                        nberr = nberr+1
                        if(nberr>hamming.to_i)
                                return -1               
                        end
                end
        end
        return nberr
    end
