                def loadIndex
                        seq = ''
                        begin
                                file = File.new(@file_suffix+FILE_SUFFIX_EXT, "r")
                                while (line = file.gets)
                                        input = line.downcase.chomp
                                        seq << input
                                end
                                file.close
                        rescue => err
                                $log.error("Exception: #{err}")
                                exit()
                        end
                        @sequence = seq
                        clear()
            @min_position = 0
                @max_position = 0
                end
