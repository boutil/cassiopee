        def extractSuffix(start,len)
                sequence = ''
                begin
                        file = File.new(@file_suffix+FILE_SUFFIX_EXT, "r")
                              file.pos = start
                                        sequence = file.read(len)
                        file.close
                rescue => err
                        puts "Exception: #{err}"
                        return nil
                end
                return sequence
        end
