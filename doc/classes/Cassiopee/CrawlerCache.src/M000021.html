                def saveCache(obj)
                        self.cache = obj
                        marshal_dump = Marshal.dump(self)
                        sfxpos = File.new(@file_suffix+FILE_CACHE_EXT,'w')
                        sfxpos = Zlib::GzipWriter.new(sfxpos)
                        sfxpos.write marshal_dump
                        sfxpos.close
                end
